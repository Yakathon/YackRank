 20 -def _send(method, url, endpoint, params, data=None): 

 21      """Return response from server after making request with method to endpoint 
 22      with params and, optionally, data""" 
 23      url = urljoin(url, endpoint) 
 24      version = settings.YIKYAK_VERSION + settings.YIKYAK_VERSION_LETTER 
 25      params += [("version", version)] 
 26      salt = str(int(time())) 
 27      msg = urljoin("/api/", endpoint) + '?' + unquote(urlencode(params)) + salt 
 28      hash_string = hash_msg(settings.YIKYAK_APIKEY, msg) 
 29      params += [("salt", salt), ("hash", hash_string)] 
 30      if method == "POST": 
 31          data = sorted(data + params) 
 32      user_agent = get_user_agent() 
 33      headers = {"User-Agent": user_agent, "Accept-Encoding": "gzip"} 
 34      return REQUEST(method, url, params=params, data=data, headers=headers)
